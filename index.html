nil vs. NULL


  • Both represent empty pointer
  • Use nil for empty Objective-C objects
  • USe NULL for empty native C-style pointers

id and Class Types



id mysteryObject = @"I'm actually an NSString";
NSLog(@"%@", [mysteryObject description]);

  • Generic object data type
  • Lose compile-time information about the object
  • Don't need the asterisk



Class targetClass = [NSString class];
id mysteryObject = [NSNumber numberWithInt:5];
NSLog([mysteryObject isKindOfClass:targetClass] ? @"YES" : @"NO");

  • Classes are objects
  • Allows for weak typing, like id does

@end

Classes


Person.m


#import "Person.h"

@implementation Person

- (instancetype)initWithName:(NSString *)name andAge:(NSNumber *)age {
    if (self = [super init]) {
        _name = name;
        _age = age;
    }
    
    return self;
}

- (void)sayHello {
    NSLog(@"Hello, %@!", self.name);
}

@end
            

Person *bob = [[Person alloc] initWithName:@"Bob" andAge:@21];
[bob sayHello];
NSLog(@"%@", bob.age);
bob.age = @23;
NSLog(@"%@", bob.age);
